Code Examples

🔷 Tailwind — Card dengan Shadow 
<!-- Card dengan shadow default -->
<div class="shadow-md p-6 rounded-xl bg-white">
  Card with medium shadow
</div>

<!-- Hover elevation effect -->
<div class="shadow-sm hover:shadow-lg transition-shadow duration-200 rounded-xl p-6">
  Elevate on hover
</div>

<!-- Modal dengan shadow maximum -->
<div class="fixed inset-0 shadow-2xl rounded-2xl bg-white">
  Full page modal
</div>

<!-- Floating panel -->
<div class="absolute top-4 right-4 shadow-xl rounded-xl bg-white p-4">
  Floating panel
</div>
🟠 CSS Variable — Token Langsung 
/* Deklarasi CSS Variables di :root */
:root {
  --shadow-none : none;
  --shadow-sm   : 0 1px 2px rgba(0,0,0,.05);
  --shadow-md   : 0 4px 6px rgba(0,0,0,.1), 0 2px 4px rgba(0,0,0,.06);
  --shadow-lg   : 0 10px 15px rgba(0,0,0,.1), 0 4px 6px rgba(0,0,0,.05);
  --shadow-xl   : 0 20px 25px rgba(0,0,0,.1), 0 10px 10px rgba(0,0,0,.04);
  --shadow-2xl  : 0 25px 50px rgba(0,0,0,.25);
}

/* Penggunaan di komponen */
.card          { box-shadow: var(--shadow-md); }
.card:hover    { box-shadow: var(--shadow-lg); }
.modal         { box-shadow: var(--shadow-2xl); }
.tooltip       { box-shadow: var(--shadow-lg); }
.input:focus   { box-shadow: var(--shadow-sm); }
⚡ Transition Shadow — Smooth Elevation 
/* Smooth elevation transition */
.interactive-card {
  box-shadow: var(--shadow-sm);
  transition: box-shadow 200ms ease, transform 200ms ease;
}
.interactive-card:hover {
  box-shadow: var(--shadow-lg);
  transform: translateY(-2px);
}

/* Tailwind equivalent */
/* class="shadow-sm hover:shadow-lg hover:-translate-y-0.5 transition-all duration-200" */
